create view vw_reportdolist 
(dodate,custnum,custname,
 scode,sname,icode,pqty,
 qty,uprice,donum,create_date,avg) as
select
		sdomas.dodate,
		custmast.custnum,
		custmast.custname,
		supmast.scode,
		supmast.sname,
		sdodet.icode,
		sdodet.pqty,
		sdodet.qty,
		sdodet.uprice,
		sdomas.donum[1],
		sdomas.create_date,
		(sdodet.pqty/sdodet.qty)		
from sdomas, supmast,sdodet, custmast
where sdomas.scode = supmast.scode
and sdomas.custnum = custmast.custnum
and sdomas.donum = sdodet.donum
and supmast.balcalc = 1

--- ISQL
create view vw_reportsolist (dodate,sname,scode,icode,desc,qty,uprice,trspt,pqty,donum,custnum,custname,avg) as 
  select x2.dodate ,x3.sname ,x3.scode ,x1.icode ,x0.desc ,x1.qty 
    ,x1.uprice ,x1.trspt ,x1.pqty ,x2.donum [1,1] ,x2.custnum 
    ,x4.custname ,(x1.pqty / x1.qty ) from itemast x0 ,sdodet 
    x1 ,sdomas x2 ,supmast x3 ,custmast x4 where 
    (((((x2.custnum = x4.custnum ) AND (x1.donum = x2.donum ) 
    ) AND (x2.scode = x3.scode ) ) AND (x0.icode = x1.icode ) 
    ) AND (x2.dostat != 'X' ) ) ;
